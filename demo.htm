jQuery • highlightTextarea • Demos
Change highlighting color
$("textarea").highlightTextarea({
    words: ["Lorem ipsum","vulputate"],
    color: "#ADF0FF"
});
Case insensitive
$("textarea").highlightTextarea({
    words: ["lOrEm IpSuM","VULPUTATE"],
    caseSensitive: false
});
Resizing (with jQueryUI)
$("textarea").highlightTextarea({
    words: ["Lorem ipsum","Nam varius","Nullam"],
    resizable: true
});
Use regular expressions (here, highlight Smarty tags)
$("textarea").highlightTextarea({
    words: ["\{(.+?)\}"]
});
Customise highlight element
$("textarea").highlightTextarea({
    words: ["Lorem ipsum","vulputate"],
    color: "#99f"
});
.highlight {
    border-radius: 0.5em;
    padding: 0 3px;
    margin: 0 -3px;
}
Use API
Disable | Enable | Change words | Change color | Debug 
<a class="disable">Disable</a> | <a class="enable">Enable</a> | <a class="setWords">Change 
words</a> | <a class="color">Change color</a> | <a class="debug">Debug</a>
$("textarea").highlightTextarea({
    words: ["Lorem ipsum","vulputate"]
});
$(".disable").click(function() {
    $("textarea").highlightTextarea('disable');
});
$(".enable").click(function() {
    $("textarea").highlightTextarea('enable');
});
$(".setWords").click(function() {
    $("textarea").highlightTextarea('setWords', [ 'Quisque','auctor' ]);
});
$(".color").click(function() {
    $("textarea").highlightTextarea('setOptions', { color: '#ADF0FF' });
});
$(".debug").click(function() {
    $("textarea").highlightTextarea('setOptions', { debug: true });
});
